// ===== Simple Increment/Decrement ===== let count = 0; const countResult = document.getElementById("countResult"); countResult.textContent = count; function countUp() { count++; countResult.textContent = count; } function countDown() { count--; countResult.textContent = count; } // ===== Compound Assignment ===== let countBy = 0; const countByResult = document.getElementById("countByResult"); function countUpBy() { const input = document.getElementById("countByValue"); const step = parseFloat(input.value); if (!isNaN(step)) { countBy += step; countByResult.textContent = countBy; input.value = ""; // clear input } else { alert("Please enter a valid number."); } } function countDownBy() { const input = document.getElementById("countByValue"); const step = parseFloat(input.value); if (!isNaN(step)) { countBy -= step; countByResult.textContent = countBy; input.value = ""; // clear input } else { alert("Please enter a valid number."); } } // ===== Arithmetic Operators ===== const arithmeticResult = document.getElementById("arithmeticResult"); function getOperands() { const op1Input = document.getElementById("op1"); const op2Input = document.getElementById("op2"); const op1 = parseFloat(op1Input.value); const op2 = parseFloat(op2Input.value); if (isNaN(op1) || isNaN(op2)) { alert("Please enter valid numbers for both operands."); return null; } return [op1, op2]; } function add() { const operands = getOperands(); if (operands) { arithmeticResult.textContent = (operands[0] + operands[1]).toFixed(2); clearArithmeticInputs(); } } function subtract() { const operands = getOperands(); if (operands) { arithmeticResult.textContent = (operands[0] - operands[1]).toFixed(2); clearArithmeticInputs(); } } function multiply() { const operands = getOperands(); if (operands) { arithmeticResult.textContent = (operands[0] * operands[1]).toFixed(2); clearArithmeticInputs(); } } function divide() { const operands = getOperands(); if (operands) { if (operands[1] === 0) { arithmeticResult.textContent = "Cannot divide by zero!"; } else { arithmeticResult.textContent = (operands[0] / operands[1]).toFixed(2); } clearArithmeticInputs(); } } function modulo() { const operands = getOperands(); if (operands) { arithmeticResult.textContent = (operands[0] % operands[1]).toFixed(2); clearArithmeticInputs(); } } function clearArithmeticInputs() { document.getElementById("op1").value = ""; document.getElementById("op2").value = ""; } // ===== Math Object Functions ===== const mathResult = document.getElementById("mathResult"); function getMathOperand() { const input = document.getElementById("mathOp"); const value = parseFloat(input.value); if (isNaN(value)) { alert("Please enter a valid number."); return null; } return value; } function pi() { mathResult.textContent = Math.PI.toFixed(4); } function sine() { const value = getMathOperand(); if (value !== null) { mathResult.textContent = Math.sin(value).toFixed(4); document.getElementById("mathOp").value = ""; } } function squareRoot() { const value = getMathOperand(); if (value !== null) { if (value < 0) { mathResult.textContent = "Cannot take square root of negative number!"; } else { mathResult.textContent = Math.sqrt(value).toFixed(4); } document.getElementById("mathOp").value = ""; } } function randomNum() { mathResult.textContent = Math.random().toFixed(4); }
